```vue{hl_lines=["6-25"]}
<script>
(... 생략 ...)
export default {
(... 생략 ...)
  methods: {
    voteQuestion(question_id) {
      if(confirm('정말로 추천하시겠습니까?')) {
        let url = "/api/question/vote";
        let params = {
          question_id: question_id
        }
        fastapi('post', url, params, () => {
            this.getQuestion()
        },
        (err_json) => {
          this.error = err_json
        })
      }
    },
    voteAnswer(answer_id) {
      if(confirm('정말로 추천하시겠습니까?')) {
        let url = "/api/answer/vote";
        let params = {
          answer_id: answer_id
        }
        fastapi('post', url, params, () => {
            this.getQuestion()
        },
        (err_json) => {
          this.error = err_json
        })
      }
    },
  },  
  created() {
    this.getQuestion();
  }
}
</script>
```
